20 static Network multi_layer_perceptron(size_t input_neurons, size_t layers,
21  size_t hidden_neurons,
22  size_t output_neurons,
23  bool is_classifier = true) {
24  NetworkBuilder builder;
25  builder.clipping(0.5)
26  .initializer(new EigenInitializer())
27  .optimizer(new Adam(0.01))
28  .layer(new Dense(new LeakyRelu(), (int)input_neurons, 0.1));
29 
30  for (size_t t = 0; t < layers; t++) {
31  builder.layer(new Dense(new LeakyRelu(), (int)hidden_neurons, 0.1))
32  }
33 
34  if (is_classifier) {
35  builder.layer(new Dense(new Softmax(), (int)output_neurons, 0.1));
36  builder.loss_function(new CategoricalCrossEntropy());
37  builder.evaluation_function(
38  {new ArgMax(), new MeanAbsolute(), new MeanSquared()});
39  } else {
40  builder.layer(new Dense(new SigmoidFunction(), (int)output_neurons, 0.1));
41  builder.loss_function(new CategoricalCrossEntropy());
42  builder.evaluation_function({new MeanAbsolute(), new MeanSquared()});
43  }
44 
45  return Network(builder);
46 }
